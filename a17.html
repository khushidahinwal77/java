        import java.util.Arrays;

        public class SecondLargestNumber {
        
            public static int findSecondLargest(int[] array) {
                // If the array has less than 2 elements, we cannot find the second largest
                if (array.length < 2) {
                    System.out.println("Array should have at least 2 numbers.");
                    return -1;
                }
        
                // Sort the array in ascending order
                Arrays.sort(array);
        
                // Return the second last element
                return array[array.length - 2];
            }
        
            public static void main(String[] args) {
                int[] array = {12, 5, 8, 20, 16, 3};
        
                int secondLargest = findSecondLargest(array);
        
                if (secondLargest != -1) {
                    System.out.println("The 2nd largest number is: " + secondLargest);
                }
            }
        }
