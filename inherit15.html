        abstract class Animal {
            abstract void sound();
            }
            
            class Lion extends Animal {
            void sound() {
            System.out.println("Roar");
            }
            }
            
            class Tiger extends Animal {
            void sound() {
            System.out.println("Growl");
            }
            }
            
            class Main {
            public static void main(String ap[]) {
            Animal lion = new Lion();
            Animal tiger = new Tiger();
            lion.sound();
            tiger.sound();
            }
            }
