        class Vehicle {
            String make;
            String model;
            int year;
            String fuelType;
            double fuelEfficiency;
            
            Vehicle(String make, String model, int year, String fuelType, double fuelEfficiency) {
            this.make = make;
            this.model = model;
            this.year = year;
            this.fuelType = fuelType;
            this.fuelEfficiency = fuelEfficiency;
            }
            double distanceTraveled(double fuel) {
            return fuel * fuelEfficiency;
            }
            }
            
            class Truck extends Vehicle {
            Truck(String make, String model, int year, String fuelType, double fuelEfficiency) {
            super(make, model, year, fuelType, fuelEfficiency);
            }
            }
            
            class Car extends Vehicle {
            Car(String make, String model, int year, String fuelType, double fuelEfficiency) {
            super(make, model, year, fuelType, fuelEfficiency);
            }
            }
            
            class Motorcycle extends Vehicle {
            Motorcycle(String make, String model, int year, String fuelType, double fuelEfficiency) {
            super(make, model, year, fuelType, fuelEfficiency);
            }
            }
